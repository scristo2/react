###component error
class ErrorComponent extends Component{

    constructor(props){

        super(props);
        this.state = {

             hasError : false,
             message : ""
        }
    }
   
    componentDidCatch(error){

        this.setState({

            hasError : true,
            message : error.message ##.message herence
        })
    }

  
    render(){

        if(this.state.hasError){

            return <h1>{this.state.message}</h1>
        
        }else{

            return this.props.children; <<<importante que devuelva el hijo que eneste ejemplo essta arriba que es title   
        }
    }
}


export default ErrorComponent;
